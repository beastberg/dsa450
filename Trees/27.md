class Solution {
  public:
    int isTree(int n, int m, vector<vector<int>> &edges) {
        if(m!=n-1)
            return false;
        vector<bool>vis(n,false);
        vector<int>adj[n]; // this is vector of array;
        
        for(int i=0;i<m;i++){
            adj[edges[i][0]].push_back(edges[i][1]);
            adj[edges[i][1]].push_back(edges[i][0]);
        }
        
        int node = 0;
        queue<int>q;
        
        vis[node] = true;
        int visited = 1;
        q.push(node);
        
        while(q.size()){
            int curr = q.front();
            q.pop();
            for(auto it:adj[curr]){
                if(!vis[it]){
                    vis[it] = true;
                    visited++;
                    q.push(it);
                }
            }
        }
        
        return (visited==n);
    }
};

**Purpose:** Performs DFS to check for cycles and connectivity.

### `isTree` Function
```cpp

```
**Purpose:** Main function to check if a graph is a tree.

**Parameters:**
- `n`: Number of nodes.
- `m`: Number of edges.
- `edges`: List of edges where each edge is a pair {u, v}.
