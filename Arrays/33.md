### minimum swap required to bring element together  
```cpp
class Solution {
  public:
    int minSwap(vector<int>& arr, int k) {
        int n = arr.size();
        int good = 0;
        
        for(int i=0;i<n;i++){
            if(arr[i]<=k)
                good++;
        }
        
        int bad = 0;
        for(int i=0;i<good;i++){
            if(arr[i]>k)
                bad++;
        }
        
        int minswap = bad;
        for(int i=good;i<n;i++){
            if(arr[i]>k)
                bad++;
            if(arr[i-good]>k)
                bad--;
            minswap = min(minswap,bad);
        }
        return minswap;
    }
};
```
